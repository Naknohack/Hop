--// Fast Attack Ultra Optimized

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Workspace = game:GetService("Workspace")

local Player = Players.LocalPlayer
local Character = Player.Character or Player.CharacterAdded:Wait()

--// Net
local Modules = ReplicatedStorage:WaitForChild("Modules")
local Net = Modules:WaitForChild("Net")

local RegisterAttack = Net:WaitForChild("RE/RegisterAttack")
local RegisterHit = Net:WaitForChild("RE/RegisterHit")
local ShootGunEvent = Net:WaitForChild("RE/ShootGunEvent")

--// Config
local ATTACK_DISTANCE = 180
local SCAN_DELAY = 0.12

--// Cache
local lastScan = 0
local cachedTarget = nil
local EnemyRootPart = nil

--// Get Character nhanh
Player.CharacterAdded:Connect(function(c)
    Character = c
end)

--// Check sống
local function IsAlive(model)
    local h = model and model:FindFirstChild("Humanoid")
    return h and h.Health > 0
end

--// Tìm target (chỉ 1)
local function GetTarget()
    local now = tick()
    if now - lastScan < SCAN_DELAY then
        return cachedTarget
    end
    lastScan = now
    cachedTarget = nil

    if not Character or not IsAlive(Character) then return end
    local myRoot = Character:FindFirstChild("HumanoidRootPart")
    if not myRoot then return end

    local pos = myRoot.Position

    local function scan(folder)
        if not folder then return end
        for _, e in ipairs(folder:GetChildren()) do
            if e ~= Character and IsAlive(e) then
                local root = e:FindFirstChild("HumanoidRootPart")
                if root then
                    if (pos - root.Position).Magnitude <= ATTACK_DISTANCE then
                        EnemyRootPart = root
                        cachedTarget = {e, root}
                        return true
                    end
                end
            end
        end
    end

    if scan(Workspace:FindFirstChild("Enemies")) then return cachedTarget end
    if scan(Workspace:FindFirstChild("Characters")) then return cachedTarget end

    return cachedTarget
end

--// Attack chính
local function Attack()
    if not Character or not IsAlive(Character) then return end

    local tool = Character:FindFirstChildOfClass("Tool")
    if not tool then return end

    local target = GetTarget()
    if not target then return end

    --// Gun
    if tool.ToolTip == "Gun" then
        tool:Activate()

        local pos = target[2].Position
        for i = 1, 2 do -- spam nhẹ
            ShootGunEvent:FireServer(pos, {target[2]})
        end
        return
    end

    --// Melee
    RegisterAttack:FireServer(0)
    RegisterHit:FireServer(target[2], {target})
end

--// Loop nhanh
RunService.RenderStepped:Connect(Attack)

--// Hook tăng tốc thêm
local mt = getrawmetatable(game)
local old = mt.__namecall
setreadonly(mt, false)

mt.__namecall = newcclosure(function(self, ...)
    if getnamecallmethod() == "FireServer" and self.Name == "RegisterAttack" then
        Attack()
        return
    end
    return old(self, ...)
end)

setreadonly(mt, true)
